                        label="Tahun Lulus"
                        name="tahun_lulus"
                        type="number"
                        value={data.tahun_lulus}
                        onChange={(e) => setData('tahun_lulus', parseInt(e.target.value, 10) || new Date().getFullYear())}
                        error={errors.tahun_lulus}
                        required
                      />
                    </div>
                  </FormSection>
                  
                  <FormSection
                    title="Pilihan Jurusan"
                    description="Pilih jurusan yang diminati. Anda dapat memilih 2 jurusan sebagai pilihan pertama dan kedua."
                    icon={<GraduationCap className="h-5 w-5" />}
                  >
                    <div className="space-y-6">
                      <div className="bg-orange-500/10 border border-orange-500/20 rounded-lg p-4">
                        <div className="flex items-start mb-3">
                          <Info className="h-5 w-5 text-orange-500 mr-2 mt-0.5 flex-shrink-0" />
                          <p className="text-sm text-white/70">
                            Pilihlah jurusan yang sesuai dengan minat dan bakat Anda. Pilihan pertama akan diutamakan dalam proses seleksi.
                          </p>
                        </div>
                      </div>
                      
                      <FormField
                        label="Pilihan Jurusan Pertama"
                        name="jurusan_1"
                        type="select"
                        value={data.jurusan_1}
                        onChange={(e) => setData('jurusan_1', e.target.value)}
                        error={errors.jurusan_1}
                        options={options.jurusan}
                        required
                      />
                      
                      <FormField
                        label="Pilihan Jurusan Kedua"
                        name="jurusan_2"
                        type="select"
                        value={data.jurusan_2}
                        onChange={(e) => setData('jurusan_2', e.target.value)}
                        error={errors.jurusan_2}
                        options={options.jurusan}
                        required
                      />
                    </div>
                  </FormSection>
                </div>
              )}
              
              {/* Step 4: Upload Dokumen */}
              {currentStep === 4 && (
                <div>
                  <FormSection
                    title="Upload Dokumen"
                    description="Upload dokumen pendukung dalam format JPG, PNG, atau PDF dengan ukuran maksimal 2MB per file."
                    icon={<Upload className="h-5 w-5" />}
                  >
                    <div className="space-y-6">
                      <div className="bg-orange-500/10 border border-orange-500/20 rounded-lg p-4 mb-6">
                        <div className="flex items-start mb-2">
                          <Info className="h-5 w-5 text-orange-500 mr-2 mt-0.5 flex-shrink-0" />
                          <p className="text-sm text-white/70">
                            Dokumen yang diunggah harus jelas, tidak buram, dan merupakan dokumen asli (bukan fotokopi dari fotokopi).
                          </p>
                        </div>
                        <ul className="text-sm text-white/70 ml-7 list-disc space-y-1">
                          <li>Pastikan ukuran file tidak melebihi 2MB</li>
                          <li>Format file yang diterima: JPG, PNG, dan PDF</li>
                          <li>Hindari menggunakan nama file dengan karakter khusus</li>
                        </ul>
                      </div>
                      
                      <div>
                        <UploadField
                          label="Pas Foto (3x4)"
                          name="pas_foto"
                          accept="image/*"
                          onChange={(e) => handleFileChange('pas_foto', e.target.files)}
                          error={errors.pas_foto}
                          progress={progress}
                          required
                        />
                        <p className="mt-1 text-xs text-white/50 ml-1">
                          Pas foto terbaru dengan latar belakang berwarna. Tipe file: PNG, JPG.
                        </p>
                      </div>
                      
                      <div>
                        <UploadField
                          label="Kartu Keluarga"
                      <UploadField
                        label="Pas Foto (3x4)"
                        name="pas_foto"
                        accept="image/*"
                        onChange={(e) => handleFileChange('pas_foto', e.target.files)}
                        error={errors.pas_foto}
                        progress={progress}
                        required
                      />
                      
                      <UploadField
                        label="Kartu Keluarga"
                        name="kartu_keluarga"
                        accept=".pdf,image/*"
                        onChange={(e) => handleFileChange('kartu_keluarga', e.target.files)}
                        error={errors.kartu_keluarga}
                        progress={progress}
                        required
                      />
                      
                      <UploadField
                        label="Akta Kelahiran"
                        name="akta_kelahiran"
                        accept=".pdf,image/*"
                        onChange={(e) => handleFileChange('akta_kelahiran', e.target.files)}
                        error={errors.akta_kelahiran}
                        progress={progress}
                        required
                      />
                      
                      <UploadField
                        label="Ijazah/Surat Keterangan Lulus"
                        name="ijazah"
                        accept=".pdf,image/*"
                        onChange={(e) => handleFileChange('ijazah', e.target.files)}
                        error={errors.ijazah}
                        progress={progress}
                        required
                      />
                      
                      <UploadField
                        label="Rapor Terakhir"
                        name="rapor"
                        accept=".pdf,image/*"
                        onChange={(e) => handleFileChange('rapor', e.target.files)}
                        error={errors.rapor}
                        progress={progress}
                        required
                      />
                      
                      <div className="bg-white/5 border border-white/10 rounded-lg p-5 mt-6">
                        <div className="flex items-center mb-4">
                          <CheckCircle className="h-5 w-5 text-orange-500 mr-2" />
                          <h3 className="font-medium text-white">Pernyataan</h3>
                        </div>
                        <p className="text-sm text-white/70 mb-4">
                          Dengan mengisi formulir ini, saya menyatakan bahwa semua informasi yang saya berikan adalah benar dan dapat dipertanggungjawabkan. Saya bersedia menerima konsekuensi jika terdapat ketidakbenaran dari informasi yang saya berikan.
                        </p>
                        <div className="flex items-center">
                          <input 
                            type="checkbox" 
                            id="agreement" 
                            className="w-4 h-4 rounded border-white/30 bg-white/5 text-orange-500 focus:ring-orange-500"
                            required
                          />
                          <label htmlFor="agreement" className="ml-2 text-sm font-medium text-white">
                            Saya menyetujui pernyataan ini
                          </label>
                        </div>
                      </div>
                    </div>
                  </FormSection>
                </div>
              )}
              
              {/* Navigation */}
              <div className="bg-white/5 backdrop-blur-sm px-6 py-4 flex items-center justify-between border-t border-white/10">
                {currentStep > 1 ? (
                  <button
                    type="button"
                    onClick={prevStep}
                    className="px-5 py-2 bg-white/10 hover:bg-white/20 border border-white/20 rounded-lg text-white font-medium transition-all duration-300"
                  >
                    Sebelumnya
                  </button>
                ) : (
                  <div></div>
                )}
                
                {currentStep < totalSteps ? (
                  <button
                    type="button"
                    onClick={nextStep}
                    className="px-5 py-2 bg-orange-500 hover:bg-orange-600 rounded-lg text-white font-medium transition-all duration-300"
                  >
                    Selanjutnya
                  </button>
                ) : (
                  <button
                    type="submit"
                    disabled={processing}
                    className="px-5 py-2 bg-orange-500 hover:bg-orange-600 rounded-lg text-white font-medium transition-all duration-300 disabled:opacity-70 disabled:cursor-not-allowed"
                  >
                    {processing ? 'Memproses...' : 'Kirim Pendaftaran'}
                  </button>
                )}
              </div>
            </form>
          </div>
        </div>
        
        <Footer />
      </div>
    </>
  );
}
